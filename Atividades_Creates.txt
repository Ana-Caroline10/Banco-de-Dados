Atividades_Creates

CREATE DATABASE Pizzaria_Apple_Rose_Grille
USE Pizzaria_Apple_Rose_Grille;

CREATE TABLE tb_enderecos (
    id_endereco INT PRIMARY KEY identity,
    logradouro VARCHAR(100),
    bairro VARCHAR(50),
    cidade_id INT,
    cep VARCHAR(8)
);

CREATE TABLE cad_cidades (
    id_cidade INT PRIMARY KEY identity,
    nome_cidade VARCHAR(50),
    uf_id INT
);

CREATE TABLE cad_uf (
    id_uf INT PRIMARY KEY identity,
    nome_uf VARCHAR(30),
    pais_id INT
);

CREATE TABLE cad_paises (
    id_pais INT PRIMARY KEY identity,
    nome_pais VARCHAR(45)
);

CREATE TABLE tb_funcionarios (
    id_funcionario INT PRIMARY KEY identity,
    nome_funcionario VARCHAR(45),
    cpf VARCHAR(11),
    matricula VARCHAR(15),
    senha VARCHAR(256),
    grupo CHAR(1),
    motorista VARCHAR(15),
    validade_motorista DATE,
    observacao VARCHAR(45),
    telefone VARCHAR(11),
    email VARCHAR(45),
    endereco_id INT,
    numero VARCHAR(10),
    complemento VARCHAR(45)
);

 CREATE TABLE tb_cargo
(id_cargo INT PRIMARY KEY IDENTITY not null,
nome_cargo VARCHAR (30),
salario INT,
funcionario_id INT
);

CREATE TABLE tb_clientes (
    id_cliente INT PRIMARY KEY IDENTITY NOT NULL,
    nome_cliente VARCHAR(50),
    cpf VARCHAR(11) UNIQUE,
    telefone VARCHAR(11),
    email VARCHAR(45),
    endereco_id INT,
    sexo INT,
    numero VARCHAR(10),
    complemento VARCHAR(45)
);

CREATE TABLE cad_ingredientes (
    id_ingrediente INT PRIMARY KEY  NOT NULL,
    descricao_ingrediente VARCHAR(100) UNIQUE,
	
);


CREATE TABLE tb_sabores (
    id_sabor INT PRIMARY KEY IDENTITY NOT NULL , 
    descricao_sabor VARCHAR(50) UNIQUE,
    foto VARBINARY(MAX),
    categoria CHAR(1),
    valor_sabor DECIMAL(11,2),
    tipo CHAR(1),
	pizzas_id INT
);



CREATE TABLE tb_sabores_ingredientes
(id_sabores_ingredientes INT PRIMARY KEY IDENTITY not null,
sabor_id INT,
ingrediente_id INT
);


CREATE TABLE cad_produtos (
    id_produto INT PRIMARY KEY identity,
    descricao_produto VARCHAR (45),
    valor DECIMAL (11,2),
    tipo CHAR (1),
    valor_unitaria VARCHAR (20)
);

CREATE TABLE tb_pizzas
(id_pizza INT PRIMARY KEY IDENTITY not null,
tamanho CHAR (3) not null,
valor_unitario decimal(11,2),

);

create table tb_pizza_sabores (
id_pizza_sabores INT PRIMARY KEY IDENTITY not null,
pizza_id INT,
sabor_id INT
);

CREATE TABLE tb_pedido
(id_pedido INT PRIMARY KEY IDENTITY not null,
cliente_id INT,
data_compra DATE,
funcionario_id INT,
quantidade INT,
valor_total DECIMAL (11,2)
);

 CREATE TABLE tb_itens_pedido_pizza
(id_intens_pedido_pizza INT PRIMARY KEY IDENTITY not null,
 quantidade INT,
 pedido_id INT,
 pizza_id INT
);


CREATE TABLE tb_itens_pedido 
(id_itens_pedido INT PRIMARY KEY IDENTITY not null,
valor DECIMAL (11,2),
quantidade INT,
produtos_id INT,
pedido_id INT 
);

CREATE TABLE tb_controle_pagamentos
(id_controle_pagamentos INT PRIMARY KEY IDENTITY not null,
descricao VARCHAR (30) not null,
pedido_id INT 
);

/* ALTER TABLE */

ALTER TABLE tb_itens_pedido_pizza ADD CONSTRAINT FK_pedidos_itens_pizza
FOREIGN KEY (pizza_id)
REFERENCES tb_pizzas (id_pizza);

ALTER TABLE tb_pedido ADD CONSTRAINT FK_pedido_funcionario
FOREIGN KEY (funcionario_id)
REFERENCES tb_funcionarios (id_funcionario);

ALTER TABLE tb_controle_pagamentos ADD CONSTRAINT  FK_controle_pagamentos_pedido
FOREIGN KEY (pedido_id)
REFERENCES tb_pedido (id_pedido);

ALTER TABLE tb_pedido ADD CONSTRAINT FK_pedido_cliente
FOREIGN KEY (cliente_id)
REFERENCES tb_clientes (id_cliente);

ALTER TABLE tb_itens_pedido_pizza ADD CONSTRAINT FK_itens_pizza_pedido
FOREIGN KEY (pedido_id)
REFERENCES tb_pedido (id_pedido);


ALTER TABLE tb_itens_pedido ADD CONSTRAINT FK_pedido_itens_pedido
FOREIGN KEY (pedido_id)
REFERENCES tb_pedido (id_pedido);

ALTER TABLE tb_itens_pedido ADD CONSTRAINT FK_Itens_pedido_produto
FOREIGN KEY (produtos_id)
REFERENCES cad_produtos (id_produto);

ALTER TABLE tb_enderecos ADD CONSTRAINT FK_endereco_cidade
FOREIGN KEY (cidade_id)
REFERENCES cad_cidades(id_cidade);

ALTER TABLE cad_cidades ADD CONSTRAINT FK_cidade_uf
FOREIGN KEY (uf_id)
REFERENCES cad_uf (id_uf);

ALTER TABLE cad_uf ADD CONSTRAINT FK_uf_pais
FOREIGN KEY (pais_id)
REFERENCES cad_paises (id_pais);

ALTER TABLE tb_funcionarios ADD CONSTRAINT FK_funcionario_endereco
FOREIGN KEY (endereco_id)
REFERENCES tb_enderecos (id_endereco);

ALTER TABLE tb_clientes ADD CONSTRAINT FK_cliente_endereco
FOREIGN KEY (endereco_id)
REFERENCES tb_enderecos (id_endereco);

ALTER TABLE tb_cargo ADD CONSTRAINT FK_cargo_funcionario
FOREIGN KEY (funcionario_id)
REFERENCES tb_funcionarios (id_funcionario);

ALTER TABLE tb_sabores_ingredientes ADD CONSTRAINT FK_itens_ingredientes
FOREIGN KEY (ingrediente_id)
REFERENCES cad_ingredientes (id_ingrediente);

ALTER TABLE tb_sabores_ingredientes ADD CONSTRAINT FK_iten_sabor
FOREIGN KEY (sabor_id)
REFERENCES tb_sabores (id_sabor);

ALTER TABLE tb_sabores ADD CONSTRAINT FK_pizzas_sabores
FOREIGN KEY (pizzas_id)
REFERENCES tb_pizzas (id_pizza);
